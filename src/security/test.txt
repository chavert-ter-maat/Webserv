server A {
		listen				8080;
		listen				8081;
		listen				8082;
		listen				8083;
		server_name			our-site.org www.our-site.org;
		error_page			404 error/404.html;
		error_page			402 /error/402.html;
		return				301 https://$host$request_url;
		allow_methods		GET POST DELETE;
		root				htmlT;
		location	/{
			allow_methods		GET POST;
			autoindex			on;
		}
		location	/uploads{
			autoindex			off;
			index				index.html;
		}
		location	/hoi{
			autoindex			on;
		}
		location	/cgi-bin{
			autoindex			on;
			root				/hoi;
		}
}
