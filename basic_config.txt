server A {
		listen				8080;
		listen				8081;
		listen				8082;
		listen				8083;
		server_name			our-site.org www.our-site.org;
		error_page			404 error/404.html;
		error_page			402 /error/402.html;
		allow_methods		GET POST;
		root				html;

		location / {
			allow_methods		GET POST;
			index				index.html;
		}
		location /uploads {
			allow_methods		GET POST DELETE;
			autoindex			on;
			root				html/uploads;
		}
		location /hoi {
			autoindex			on;
		}
		location /cgi-bin {
			autoindex			on;
			root				html/cgi-bin;
		}
}
